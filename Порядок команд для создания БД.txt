Если пароль на базу данных не установлен, то можно зайти с помощью sudo mysql
Установка пароля root:
ALTER USER 'user-name'@'localhost' IDENTIFIED BY 'NEW_USER_PASSWORD';
FLUSH PRIVILEGES;

Если пароль установлен, то заходим с помощью mysql -u root -p

Создание базы данных
create database logistics_db character set utf8 collate utf8_general_ci;

Создание таблицы roles
create table roles(role_id int(5) not null auto_increment, role_name varchar(100), description varchar(250), primary key(role_id));

INSERT INTO roles(role_name,description) VALUES("sysadmin","Системный администратор. Создает любого пользователя. Имеет набор полных прав.");
INSERT INTO roles(role_name,description) VALUES("admin","Администратор.Создает, редактирует, удаляет: администраторов, клиентов, организации, мероприятия, структуру папок для документов, админит документы");
INSERT INTO roles(role_name,description) VALUES("inspector","Инспектор. Имеет право сформировать отчет: 'Анализ полученных документов', только по разрешенному Администратором мероприятия");
INSERT INTO roles(role_name,description) VALUES("participant","Участник. Download - Upload документов только из своей папки");


create table users(user_id int(5) not null auto_increment, phone_number varchar(100) not null, second_name varchar(100) not null, first_name varchar(100) not null, last_name varchar(100), full_name varchar(300), role_id int(50),        email varchar(100), city varchar(100), password varchar(100), comment varchar(500), disabled bool, primary key (user_id), foreign key(role_id) references roles(role_id));

create table participants(participant_id int(5) not null auto_increment, phone_number varchar(100) not null, second_name varchar(100), first_name varchar(100), last_name varchar(100), full_name varchar(300), role_id int(50), email varchar(100), city varchar(100), password varchar(100), comment varchar(500), disabled bool, primary key(participant_id), foreign key(role_id) references roles(role_id));


Основные запросы для базы данных:
0)Подключение к базе данных:
  use logistics_db;
1)Показать всю таблицу 
  select*from table_name; 
2)Вставить в таблицу
  insert into table_name(column1,column2, ...) VALUES(value1,value2, ...);
3)Выбор конкретной строки
  select id from table_name where phone_number = "phone_number"
3)Удаление по какому-то значению
  delete from table_name where column_name = value;
